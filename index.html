#画一个正六边形 import turtle#导入tle库 turtle.shape('turtle')#设置笔形状 turtle.speed(0)#设置速度 turtle.bgcolor('black') turtle.pencolor('red') turtle.pensize(3) colors=['red','green','white','blue','yellow','pink'] for i in range(0,200,1): turtle.forward(i)#往前走 默认头右转 turtle.left(61)#左转 turtle.pencolor(colors[i%6]) turtle.done()#阻止窗口关闭
